<script>
const API_BASE = "https://a-ten-kappa-21.vercel.app/";

// Recomandări AI
async function getRecommendations() {
    const query = document.getElementById("queryInput").value.trim();
    const container = document.getElementById("recommendationsContainer");
    if(!query){ container.innerHTML = "<p>Introduceți câteva detalii pentru recomandări.</p>"; return; }
    container.innerHTML = "<p>Se generează recomandările...</p>";
    try {
        const response = await fetch(`${API_BASE}/recommend`, {
            method:"POST",
            headers: {"Content-Type":"application/json"},
            body: JSON.stringify({ query })
        });
        if(!response.ok) throw new Error(`Eroare server: ${response.status}`);
        const data = await response.json();
        container.innerHTML = "";
        if(data.recommendations && data.recommendations.length){
            data.recommendations.forEach(item => {
                const p = document.createElement("p");
                p.textContent = item;
                container.appendChild(p);
            });
        } else {
            container.innerHTML = "<p>Nu s-au găsit recomandări pentru criteriile tale.</p>";
        }
    } catch(err){
        console.error(err);
        container.innerHTML = "<p>Eroare la server. Încearcă mai târziu.</p>";
    }
}
document.getElementById("askBtn").addEventListener("click", getRecommendations);

// Chat AI
document.getElementById("chatBtn").addEventListener("click", async () => {
    const question = document.getElementById("chatInput").value.trim();
    const container = document.getElementById("chatContainer");
    if(!question){ container.innerHTML = "<p>Scrie o întrebare.</p>"; return; }

    const pUser = document.createElement("p");
    pUser.textContent = `Întrebare: ${question}`;
    container.appendChild(pUser);

    const pLoading = document.createElement("p");
    pLoading.textContent = "AI răspunde...";
    container.appendChild(pLoading);

    try {
        const res = await fetch(`${API_BASE}/chat`, {
            method:"POST",
            headers: {"Content-Type":"application/json"},
            body: JSON.stringify({ question })
        });
        const data = await res.json();
        pLoading.textContent = `Răspuns AI: ${data.answer}`;
    } catch(err){
        pLoading.textContent = "Eroare la server. Încearcă mai târziu.";
    }
    document.getElementById("chatInput").value = "";
});
</script>
