<script>
const API_BASE = "https://casa-smart-beckend-new.vercel.app/api";

// Recomandări AI
async function getRecommendations() {
    const query = document.getElementById("queryInput").value.trim();
    const container = document.getElementById("recommendationsContainer");
    if(!query){ container.innerHTML = "<p>Introduceți câteva detalii.</p>"; return; }
    container.innerHTML = "<p>Se generează recomandările...</p>";
    try {
        const response = await fetch(`${API_BASE}/recommend`, {
            method:"POST",
            headers: {"Content-Type":"application/json"},
            body: JSON.stringify({ query })
        });
        if(!response.ok) throw new Error(`Eroare server: ${response.status}`);
        const data = await response.json();
        container.innerHTML = "";
        if(data.recommendations.length){
            data.recommendations.forEach(item => {
                const p = document.createElement("p");
                p.textContent = item;
                container.appendChild(p);
            });
        } else container.innerHTML = "<p>Nu s-au găsit recomandări.</p>";
    } catch(err){ console.error(err); container.innerHTML = "<p>Eroare la server.</p>"; }
}
document.getElementById("askBtn").addEventListener("click", getRecommendations);

// Login
document.getElementById("loginBtn").addEventListener("click", async () => {
    const username = document.getElementById("loginUsername").value;
    const password = document.getElementById("loginPassword").value;
    const msg = document.getElementById("loginMessage");
    try {
        const res = await fetch(`${API_BASE}/login`, {
            method:"POST",
            headers: {"Content-Type":"application/json"},
            body: JSON.stringify({ username, password })
        });
        const data = await res.json();
        msg.textContent = data.message || "Login reușit";
    } catch(err){ msg.textContent = "Autentificare eșuată"; }
});

// Register
document.getElementById("regBtn").addEventListener("click", async () => {
    const username = document.getElementById("regUsername").value;
    const password = document.getElementById("regPassword").value;
    const msg = document.getElementById("regMessage");
    try {
        const res = await fetch(`${API_BASE}/register`, {
            method:"POST",
            headers: {"Content-Type":"application/json"},
            body: JSON.stringify({ username, password })
        });
        const data = await res.json();
        msg.textContent = data.message || "Înregistrare reușită";
    } catch(err){ msg.textContent = "Înregistrare eșuată"; }
});
